"Atmosphere" is a Virtual Worlds viewer for the WWW. Lots of fun! It's like VMRL (Virt. Reality Modeling Lang), but faster, with videogame-like controls, multiple users with avatars, hooks for Javascript, 3D animation etc. Hang out with other users in 3D forums. Make virtual meeting halls, museums, even construct online multiuser games. The world-builder is currently freeware.

// I've pared Brandyn's DRAGONFLY javascript down to its bare essentials.
// Use this to add animated segments to your avatars.      - amasci
// The full DRAGONFLY has more cool features, and is at
// http://sifter.org/~brandyn/worlds/Dragonfly/dragonfly.js

if (context == remoteAvatarContext) {
   //authorUrl = "http://sifter.org/~brandyn/"
   debug = false;

   //-------------------- Dragonfly ------------------
   function initObjs()
   {
      ob = mySubWorld;
      ob.parts = [];

// First animated segment  (add your own)
      ob.parts[0] =SubWorld("./easel1.aer");
      ob.parts[0].add();
      ob.parts[0].transform = ob.transform;   //init postn
//    ob.parts[0].transparent = true;
//    ob.parts[0].visible = false;

// Another animated segment  (add your own)
//    ob.parts[1] = SubWorld("http://www.adobe.com/products/atmosphere/worlds/ChezWhat/spider-1.aer");
//    ob.parts[1].add();
//    ob.parts[1].transparent=true;
//    ob.parts[1].transform = ob.transform;   //init postn


// **********************
// Animator section
// **********************
      ob.timestep = function(now, delt)
      {
         p1 = Math.PI * 2.0;
//  change the segment position using Vector(), then rotate it constantly
         ob.parts[1].transform = Transform(ob.transform, 
            Rotation('Y', now * P1 * .4 + 3.14),
            Vector(4.3,-5,0)
            );
      }

      addAnimator(ob);
   }

   // Run the above
   initObjs();
}
